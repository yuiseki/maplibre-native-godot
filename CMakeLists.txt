cmake_minimum_required(VERSION 3.24)

project(maplibre-native-godot-poc LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

set(GODOT_CPP_DIR "${CMAKE_SOURCE_DIR}/third_party/godot-cpp" CACHE PATH "Path to a godot-cpp checkout")
set(MLN_SOURCE_DIR "" CACHE PATH "Path to maplibre-native source (required)")

if(WIN32)
  set(MLN_BUILD_DIR "${CMAKE_SOURCE_DIR}/build/maplibre-native-windows" CACHE PATH "Path to maplibre-native Windows build output")
else()
  set(MLN_BUILD_DIR "${CMAKE_SOURCE_DIR}/build/maplibre-native-linux" CACHE PATH "Path to maplibre-native build output")
endif()

if(NOT MLN_SOURCE_DIR OR NOT EXISTS "${MLN_SOURCE_DIR}/CMakeLists.txt")
  message(FATAL_ERROR
    "MLN_SOURCE_DIR must point to a maplibre-native checkout.\n"
    "  cmake ... -DMLN_SOURCE_DIR=/path/to/maplibre-native\n"
    "  Or export MLN_SOURCE_DIR before running the build scripts.")
endif()

if(NOT EXISTS "${GODOT_CPP_DIR}/CMakeLists.txt")
  message(FATAL_ERROR
    "godot-cpp was not found at ${GODOT_CPP_DIR}.\n"
    "Check out godot-cpp there before building the extension:\n"
    "  git clone https://github.com/godotengine/godot-cpp third_party/godot-cpp")
endif()

# ---------------------------------------------------------------------------
# Platform-specific maplibre-native integration
# ---------------------------------------------------------------------------

if(WIN32)
  # On Windows: import the CMake targets generated by the maplibre-native build.
  # This handles all transitive dependencies (wgpu, vcpkg libs, system libs) automatically.
  set(_mln_targets "${MLN_BUILD_DIR}/MapboxCoreTargets.cmake")
  if(NOT EXISTS "${_mln_targets}")
    message(FATAL_ERROR
      "MapboxCoreTargets.cmake not found at ${_mln_targets}.\n"
      "Run scripts\\build_maplibre_native_windows.bat first, or set MLN_BUILD_DIR\n"
      "to an existing maplibre-native Windows build directory (e.g. the slint project's\n"
      "build-ninja/vendor/maplibre-native).")
  endif()

  # Find vcpkg-installed packages referenced transitively by mbgl-core.
  # CMAKE_PREFIX_PATH must include the vcpkg_installed directory (set by the
  # build script via -DCMAKE_PREFIX_PATH or vcpkg toolchain).
  find_package(OpenGL REQUIRED)
  find_package(dlfcn-win32 CONFIG REQUIRED)
  find_package(libuv CONFIG REQUIRED)
  find_package(ICU COMPONENTS i18n uc data QUIET)
  find_package(PNG REQUIRED)

  include("${_mln_targets}")

else()
  # Linux: validate pre-built static libraries produced by build_maplibre_native_linux.sh
  set(MLN_CORE_LIB "${MLN_BUILD_DIR}/libmbgl-core.a" CACHE FILEPATH "Path to libmbgl-core.a")

  if(NOT EXISTS "${MLN_CORE_LIB}")
    message(FATAL_ERROR
      "maplibre-native core library was not found at ${MLN_CORE_LIB}.\n"
      "Run scripts/build_maplibre_native_linux.sh first.")
  endif()
endif()

# ---------------------------------------------------------------------------
# GDExtension shared library
# ---------------------------------------------------------------------------

add_subdirectory("${GODOT_CPP_DIR}" "${CMAKE_BINARY_DIR}/godot-cpp")

add_library(maplibre_native_godot SHARED
  src/register_types.cpp
  src/map_view_node.cpp
  src/map_runtime.cpp
)

target_include_directories(maplibre_native_godot PRIVATE
  "${CMAKE_SOURCE_DIR}/src"
  "${MLN_SOURCE_DIR}/include"
  "${MLN_SOURCE_DIR}/platform/default/include"
  # Vendor headers transitively required by the public maplibre-native headers
  "${MLN_SOURCE_DIR}/vendor/maplibre-native-base/include"
  "${MLN_SOURCE_DIR}/vendor/maplibre-native-base/deps/variant/include"
  "${MLN_SOURCE_DIR}/vendor/maplibre-native-base/deps/geometry.hpp/include"
  "${MLN_SOURCE_DIR}/vendor/maplibre-native-base/deps/geojson.hpp/include"
  "${MLN_SOURCE_DIR}/vendor/maplibre-native-base/deps/geojson-vt-cpp/include"
  "${MLN_SOURCE_DIR}/vendor/boost/include"
  "${MLN_SOURCE_DIR}/vendor/rapidjson/include"
  "${MLN_SOURCE_DIR}/vendor/unordered_dense/include"
  "${MLN_SOURCE_DIR}/vendor/eternal/include"
  "${MLN_SOURCE_DIR}/vendor/protozero/include"
  "${MLN_SOURCE_DIR}/vendor/webgpu-cpp"
  "${MLN_SOURCE_DIR}/vendor/webgpu-cpp/wgpu-native"
  "${MLN_SOURCE_DIR}/vendor/wgpu-native/ffi"
)

# ---------------------------------------------------------------------------
# Platform-specific linking
# ---------------------------------------------------------------------------

if(WIN32)
  target_compile_definitions(maplibre_native_godot PRIVATE
    NOMINMAX
    _USE_MATH_DEFINES
    WIN32_LEAN_AND_MEAN)

  if(MSVC)
    target_compile_options(maplibre_native_godot PRIVATE /utf-8 /MP)
  endif()

  target_link_libraries(maplibre_native_godot PRIVATE
    godot-cpp
    mbgl-core
  )

  # Copy wgpu_native.dll to bin/ so Godot can find it at runtime
  set(_wgpu_dll "${MLN_SOURCE_DIR}/vendor/wgpu-native/target/release/wgpu_native.dll")
  if(NOT EXISTS "${_wgpu_dll}")
    # Fallback: look next to the .dll.lib import library recorded in MapboxCoreTargets
    get_target_property(_wgpu_iface mbgl-vendor-wgpu INTERFACE_LINK_LIBRARIES)
    string(REGEX MATCH "[^;]*wgpu_native\\.dll\\.lib" _wgpu_lib "${_wgpu_iface}")
    if(_wgpu_lib)
      string(REPLACE ".dll.lib" ".dll" _wgpu_dll "${_wgpu_lib}")
    endif()
  endif()
  if(EXISTS "${_wgpu_dll}")
    add_custom_command(TARGET maplibre_native_godot POST_BUILD
      COMMAND ${CMAKE_COMMAND} -E copy_if_different
        "${_wgpu_dll}"
        "${CMAKE_SOURCE_DIR}/bin/wgpu_native.dll"
      COMMENT "Copying wgpu_native.dll to bin/")
  else()
    message(WARNING
      "wgpu_native.dll not found at ${_wgpu_dll}.\n"
      "Copy it manually to bin/ before running Godot.")
  endif()

elseif(UNIX AND NOT APPLE)
  # Linux: static libs linked with --whole-archive so that self-registering
  # factory objects are not stripped by the GNU linker.
  set(MLN_VENDOR_LIBS
    "${MLN_BUILD_DIR}/libmbgl-core.a"
    "${MLN_BUILD_DIR}/libmbgl-freetype.a"
    "${MLN_BUILD_DIR}/libmbgl-harfbuzz.a"
    "${MLN_BUILD_DIR}/libmbgl-vendor-csscolorparser.a"
    "${MLN_BUILD_DIR}/libmbgl-vendor-nunicode.a"
    "${MLN_BUILD_DIR}/libmbgl-vendor-sqlite.a"
    "${MLN_BUILD_DIR}/libmbgl-vendor-parsedate.a"
    "${MLN_BUILD_DIR}/vendor/maplibre-tile-spec/cpp/libmlt-cpp.a"
    "${MLN_SOURCE_DIR}/vendor/wgpu-native/target/x86_64-unknown-linux-gnu/release/libwgpu_native.a"
  )

  target_link_libraries(maplibre_native_godot PRIVATE
    godot-cpp
    -Wl,--whole-archive
    ${MLN_VENDOR_LIBS}
    -Wl,--no-whole-archive
    # System libs linked dynamically (not built with -fPIC by us)
    dl m pthread ssl crypto curl uv png jpeg webp bz2 z
    icuuc icudata icui18n
    # GPU / display
    GL X11
    # Vulkan (required by wgpu-native)
    vulkan
  )
endif()

# ---------------------------------------------------------------------------
# Output naming (matches Godot GDExtension conventions)
# ---------------------------------------------------------------------------

if(WIN32)
  if(CMAKE_BUILD_TYPE STREQUAL "Release" OR CMAKE_BUILD_TYPE STREQUAL "RelWithDebInfo")
    set(MLNG_OUTPUT_NAME "maplibre_native_godot.windows.template_release.x86_64")
  else()
    set(MLNG_OUTPUT_NAME "maplibre_native_godot.windows.template_debug.x86_64")
  endif()
  set_target_properties(maplibre_native_godot PROPERTIES
    LIBRARY_OUTPUT_DIRECTORY "${CMAKE_SOURCE_DIR}/bin"
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_SOURCE_DIR}/bin"
    PREFIX ""
    OUTPUT_NAME "${MLNG_OUTPUT_NAME}"
  )
else()
  if(CMAKE_BUILD_TYPE STREQUAL "Release" OR CMAKE_BUILD_TYPE STREQUAL "RelWithDebInfo")
    set(MLNG_OUTPUT_NAME "maplibre_native_godot.linux.template_release.x86_64")
  else()
    set(MLNG_OUTPUT_NAME "maplibre_native_godot.linux.template_debug.x86_64")
  endif()
  set_target_properties(maplibre_native_godot PROPERTIES
    LIBRARY_OUTPUT_DIRECTORY "${CMAKE_SOURCE_DIR}/bin"
    RUNTIME_OUTPUT_DIRECTORY "${CMAKE_SOURCE_DIR}/bin"
    PREFIX "lib"
    OUTPUT_NAME "${MLNG_OUTPUT_NAME}"
  )
endif()
